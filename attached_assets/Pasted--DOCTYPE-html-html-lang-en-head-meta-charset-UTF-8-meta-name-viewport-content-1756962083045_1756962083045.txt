        function showEmailSentModal() {
            const email = document.getElementById('email').value;
            
            if (!email || email.length < 4) {
                alert('Please enter a valid email address');
                return;
            }
            
            // Show the modal
            document.getElementById('modalOverlay').classList.remove('hidden');
        }

        function closeModal() {
            document.getElementById('modalOverlay').classList.add('hidden');
        }

        function openEmailApp() {
            console.log('Opening email app...');
            // In a real app, this might open the default email app
            // or provide instructions for checking email
            closeModal();
        }

        // Close modal when clicking outside of it
        document.getElementById('modalOverlay').addEventListener('click', function(e) {
            if (e.target === this) {
                closeModal();
            }
        });

        // Allow only valid email format
        document.getElementById('email').addEventListener('input', function(e) {
            const email = this.value;
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            
            if (email && !emailRegex.test(email)) {
                this.style.borderColor = '#ff6b6b';
            } else {
                this.style.borderColor = 'rgb(70, 130, 180)';
            }
        });
