    <script>
        function showModal(message) {
            document.getElementById('modal-message').textContent = message;
            document.getElementById('modal').classList.remove('hidden');
        }

        document.getElementById('modal-close').addEventListener('click', function() {
            document.getElementById('modal').classList.add('hidden');
        });

        function goToNextScreen() {
            // Add a simple animation to the button
            const button = document.querySelector('.next-button');
            button.style.transform = 'scale(0.95)';
            
            setTimeout(() => {
                button.style.transform = 'scale(1)';
                showModal('Moving to next onboarding screen...');
            }, 150);
            
            // In a real app, this would navigate to the next onboarding screen
            // or to the login options if this is the last screen
        }

        // Add touch feedback for mobile
        document.querySelector('.next-button').addEventListener('touchstart', function() {
            this.style.transform = 'scale(0.95)';
        });

        document.querySelector('.next-button').addEventListener('touchend', function() {
            this.style.transform = 'scale(1)';
        });

        // Simulate page progression (for demo purposes)
        let currentPage = 0;
        const totalPages = 3;

        function updatePageIndicators() {
            const indicators = document.querySelectorAll('.page-indicator');
            indicators.forEach((indicator, index) => {
                if (index === currentPage) {
                    indicator.classList.add('active');
                    indicator.classList.remove('inactive');
                } else {
                    indicator.classList.add('inactive');
                    indicator.classList.remove('active');
                }
            });
        }

        // For demo: simulate advancing through pages
        function simulatePageAdvancement() {
            currentPage = (currentPage + 1) % totalPages;
            updatePageIndicators();
            
            if (currentPage === 0) {
                // Last page reached, would normally go to login options
                showModal('Onboarding complete! Navigating to login options...');
            }
        }

        // Optional: Auto-advance demo (remove in real app)
        // setInterval(simulatePageAdvancement, 5000);
