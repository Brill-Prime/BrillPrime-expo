  <script>
    // Initialize the map with responsive view
    const map = L.map('map', {
      zoomControl: false, // Disable default zoom controls
      attributionControl: false
    }).setView([9.0765, 7.3986], 13); // Default to Jos, Nigeria

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>'
    }).addTo(map);

    // Handle window resize for responsiveness
    window.addEventListener('resize', () => {
      map.invalidateSize();
    });

    // Dynamic energy percentage (simulated)
    let energyLevel = 75;
    const energyPercentageElement = document.getElementById('energyPercentage');

    // Simulate energy level changes
    setInterval(() => {
      energyLevel = Math.max(0, Math.min(100, energyLevel + (Math.random() * 2 - 1)));
      energyPercentageElement.textContent = `${Math.round(energyLevel)}%`;
    }, 2000);

    // Tab functionality
    document.querySelectorAll('.tab').forEach(tab => {
      tab.addEventListener('click', () => {
        document.querySelectorAll('.tab').forEach(t => t.classList.remove('on-delivery'));
        tab.classList.add('on-delivery');
      });
    });

    // View orders button
    document.getElementById('viewOrdersButton').addEventListener('click', () => {
      alert('Viewing orders...');
    });

    // Back button functionality
    document.getElementById('backButton').addEventListener('click', () => {
      alert('Going back...');
      // In a real app, this would navigate to the previous screen
    });

    // Menu button functionality
    const menuButton = document.getElementById('menuButton');
    const menuDropdown = document.getElementById('menuDropdown');

    menuButton.addEventListener('click', () => {
      menuDropdown.classList.toggle('active');
    });

    // Close menu when clicking outside
    document.addEventListener('click', (event) => {
      if (!menuButton.contains(event.target) && !menuDropdown.contains(event.target)) {
        menuDropdown.classList.remove('active');
      }
    });

    // Menu item functionality
    document.querySelectorAll('.menu-item').forEach(item => {
      item.addEventListener('click', () => {
        alert(`Navigating to ${item.textContent}...`);
        menuDropdown.classList.remove('active');
      });
    });
  </script>
